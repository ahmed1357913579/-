        localStorage.setItem('members', JSON.stringify(members));

    }
    
    function msg(txt) {
        let d = document.createElement('div');
        d.className = 'toast';
        d.innerHTML = txt;
        document.body.appendChild(d);
        setTimeout(() => d.remove(), 2000);
    }
    
    function goTo(page) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('show'));
        document.getElementById(page).classList.add('show');
        if(page === 'members') showMembers();
        if(page === 'loans') showLoans();
        if(page === 'stats') showStats();
        if(page === 'admin') fillSelects();
    }
    
    function showMembers() {
        let search = document.getElementById('search').value.toLowerCase();
        let f = members.filter(m => m.name.toLowerCase().includes(search));
        let html = '';
        f.forEach(m => {
            let debt = 0;
            loans.forEach(l => { if(l.mid === m.id) debt += l.remain; });
            let withdrawal = m.paid - debt;
            html += `<div class="card">
                <div class="flex">
                    <div><div class="name">${m.name}</div><div class="small">📅 ${m.date} | 📞 ${m.phone}</div></div>
                    <div><div class="green">💰 ${m.paid.toLocaleString()}</div><div class="orange">🏦 ${debt.toLocaleString()}</div><div>💵 ${withdrawal.toLocaleString()}</div></div>
                </div>
                <button class="btn-dark" onclick="openWhatsapp('${m.name}','${m.phone}')">📱 طلب قرض</button>
            </div>`;
        });
        document.getElementById('membersList').innerHTML = html || '<p style="text-align:center">لا توجد نتائج</p>';
    }
    
    function showLoans() {
        let active = loans.filter(l => l.remain > 0);
        if(active.length === 0) { document.getElementById('loansList').innerHTML = '<p style="text-align:center">لا توجد قروض</p>'; return; }
        let html = '';
        active.forEach(l => {
            let p = ((l.total - l.remain) / l.total) * 100;
            html += `<div class="card">
                <div class="name" style="color:#ffc107">${l.name}</div>
                <div style="background:#333; border-radius:10px; margin:10px 0; height:6px"><div style="background:#ffc107; width:${p}%; height:6px; border-radius:10px"></div></div>
                <div class="flex"><span>💰 ${l.total.toLocaleString()}</span><span>📅 ${l.month}</span></div>
                <div class="flex"><span style="color:#ff9800">⚡ ${l.remain.toLocaleString()}</span><span>📆 ${l.due}</span></div>
            </div>`;
        });
        document.getElementById('loansList').innerHTML = html;
    }
    
    function showStats() {
        let totalPaid = 0; members.forEach(m => totalPaid += m.paid);
        let totalRemain = 0; loans.forEach(l => totalRemain += l.remain);
        let avail = totalPaid - totalRemain - 9988;
        document.getElementById('totalBal').innerHTML = totalPaid.toLocaleString();
        document.getElementById('available').innerHTML = avail.toLocaleString();
        document.getElementById('totalLoan').innerHTML = totalRemain.toLocaleString();
        document.getElementById('memberCount').innerHTML = members.length;
    }

        document.getElementById('admin1').value = admins.a1;
        document.getElementById('admin2').value = admins.a2;

    }
    
    function addSub() {
        let month = prompt('أدخل الشهر (مثال: أبريل 2026)');
        if(month) { members.forEach(m => m.paid += m.sub); save(); showMembers(); showStats(); msg('تم تسجيل اشتراكات '+month); }
    }
    
    function doPay() {
        let mid = document.getElementById('paySelect').value;
        let amount = document.getElementById('payAmount').value;
        if(!mid) { msg('اختر العضو'); return; }
        if(!amount || amount<=0) { msg('أدخل المبلغ'); return; }
        let loan = loans.find(l => l.mid == mid && l.remain > 0);
        if(!loan) { msg('لا يوجد قرض نشط'); return; }
        loan.remain = Math.max(0, loan.remain - parseFloat(amount));
        save(); showLoans(); showStats(); msg('تم تسديد '+amount+' ريال');
        document.getElementById('payAmount').value = '';
    }
    
    function addMember() {
        let name = document.getElementById('newName').value;
        let phone = document.getElementById('newPhone').value;
        if(!name || !phone) { msg('أدخل الاسم والجوال'); return; }
        let newId = Math.max(...members.map(m=>m.id),0)+1;
        let today = new Date().toISOString().split('T')[0];
        members.push({id:newId, name:name, phone:phone, date:today, paid:0, sub:200});
        save(); showMembers(); showStats(); fillSelects(); msg('تم إضافة '+name);
        document.getElementById('newName').value = '';
        document.getElementById('newPhone').value = '';
    }
    
    function addLoan() {
        let mid = document.getElementById('loanSelect').value;
        let total = document.getElementById('loanTotal').value;
        let month = document.getElementById('loanMonth').value;
        if(!mid || !total || !month) { msg('املأ الحقول'); return; }
        let m = members.find(m => m.id == mid);
        let newId = Math.max(...loans.map(l=>l.id),0)+1;
        let next = new Date(); next.setMonth(next.getMonth()+1);
        let due = next.toISOString().split('T')[0];
        loans.push({id:newId, mid:parseInt(mid), name:m.name, total:parseFloat(total), month:parseFloat(month), remain:parseFloat(total), due:due});
        save(); showLoans(); showStats(); msg('تم إضافة قرض بقيمة '+total);
        document.getElementById('loanTotal').value = '';
        document.getElementById('loanMonth').value = '';
    }
